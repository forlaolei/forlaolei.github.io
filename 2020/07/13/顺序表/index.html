顺序表

Posted on 2020-07-13 In c/c++数据结构 
#include<stdio.h>
#include<stdlib.h>
#define InitSize 10
#define true 1
#define false 0

typedef int ElemType;
typedef struct {
    // 顺序表元素
    ElemType data[InitSize];
    // 顺序表当前长度
    int length;
}SeqList;
// 初始化顺序表
void InitList(SeqList &L)
{
    L.length = 0;
}
// 顺序表插入操作 i表示第几个元素（角标+1）
bool ListInsert(SeqList &L, int i, ElemType e)
{
    if (i<1 || i>L.length + 1)
    {
        return 0;
    }
    if (L.length >= InitSize)
    {
        return 0;
    }
    for (int j = L.length; j >= i; j--)
    {
        L.data[j] = L.data[j - 1];
    }
    L.data[i - 1] = e;
    L.length++;
    return 1;
}
// 求表长
int Length(SeqList L)
{
    return L.length;
}
// 查找元素 返回是第几个元素（角标+1）
int LocateElem(SeqList L, ElemType e)
{
    for (int i = 0; i < L.length; i++)
    {
        if (e == L.data[i])
        {
            return i + 1;
        }
    }
    return 0;
}

// 查找第i个元素
ElemType GetElem(SeqList L, int i)
{
    if (i > L.length + 1)
    {
        return 0;
    }
    else
    {
        for (int j = 0; j < L.length; j++)
        {
            if (j == i - 1) {
                return L.data[j];
            }
        }
    }
    return 0;
}
// 打印顺序表
void PrintfList(SeqList L)
{
    printf("\n顺序表最大容量为：%d\n", InitSize);
    for (int i = 0; i < L.length; i++) {
        printf("data=%d\n", L.data[i]);
    }
}

// 删除第i个元素，并用e返回删除位置的值
bool ListDelete(SeqList &L, int i, ElemType &e)
{
    if (i < 1 || i > L.length)
    {
        return 0;
    }
    // 被删除的元素赋给e
    e = L.data[i - 1];
    // 删除元素后面的元素前移 角表为i-1的元素已经删除 
    // 将从角标为i开始的元素前移 总共前移L.length-i 个元素
    for (int j = i; j < L.length; j++)
    {
        L.data[j - 1] = L.data[j];
    }
    L.length--;
    return 1;
}

// 判断顺序表是否为空
const char* Empty(SeqList L)
{
    const char* str;
    if (L.length <= 0) {
        str = "顺序表为空";
        return str;
    }
    str = "顺序表非空";
    return str;
}

// 销毁顺序表
void DestroyList(SeqList &L)
{
    if (L.length == 0) {
        return;
    }
    L.length == 0;
}
int main()
{
    SeqList L;
    InitList(L);printf("%s\n", Empty(L));ElemType e1 = 1;
    ListInsert(L, 1, e1);
    ElemType e2 = 2;
    ListInsert(L, 2, e2);

    int locate = LocateElem(L, 2);
    printf("2是第%d个元素\n", locate);

    printf("第1个元素是=%d\n", GetElem(L, 1));

    ElemType deleteElem;
    ListDelete(L, 1, deleteElem);
    printf("删除的元素是=%d\n", deleteElem);

    int length = Length(L);
    printf("表长=%d\n", length);

    PrintfList(L);
    printf("%s\n", Empty(L));
    DestroyList(L);
    getchar();
    return 0;
}
